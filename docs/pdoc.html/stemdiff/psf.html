126def circular_mask(w, h, center=None, radius=None):
127    '''
128    Create a circular mask for rectangular array
129
130    Parameters
131    ----------
132    h : int
133        Height of the array.
134    w : int
135        Width of the array.
136    center : list or tuple of two integers, optional, default is None
137        Center of the mask.
138        If None, center will be a geometric center of array.
139    radius : int, optional, default is None
140        Radius of the mask.
141        If None, radius will be the distance to the shorter array wall. 
142
143    Returns
144    -------
145    mask : 2D-array of bool values
146        Circular mask: arr * mask = arr with zero values outside the mask.
147        
148    Note
149    ----
150    Copy+pasted and slightly modified from www.
151    GoogleSearch: numpy create circular mask on rectangular array
152    https://stackoverflow.com/q/44865023
153    '''
154    # Determine center and radius, if arguments were None.
155    if center is None: 
156        # use the middle of the image
157        center = (int(h/2), int(w/2))
158    if radius is None:
159        # use the smallest distance between the center and image walls
160        radius = min(center[0], center[1], h-center[0], w-center[1])
161    # Create orthogonal grid and calculate distance from center
162    # (exmplanation => https://stackoverflow.com/q/44865023
163    Y, X = np.ogrid[:w, :h]
164    dist_from_center = np.sqrt((X - center[0])**2 + (Y-center[1])**2)
165    # Calculate mask
166    # (explanation => https://stackoverflow.com/q/44865023
167    mask = dist_from_center <= radius
168    # Return the mask
169    # (2D-boolean array to modify original array
170    # (arr * mask = array with zero values outsid the mask
171    return(mask)

175    def get_psf(SDATA, DIFFIMAGES, df, circular=False):
176        '''
177        Get PSF of type1 = estimated from datafiles with low/no diffractions.
178
179        Parameters
180        ----------
181        SDATA : stemdiff.gvars.SourceData object
182            The object describes source data (detector, data_dir, filenames).
183        DIFFIMAGES : stemdiff.gvars.DiffImages object
184            Object describing the diffraction images/patterns.
185        df : pandas.DataFrame object
186            Database with datafile names and characteristics.
187            PSF is estimated from the files included in this database.
188            Typically, df is a sub-database containing low-diffracting files.
189        circular : bool, optional, default is False
190            If True, apply circular mask - change square PSF to circular PSF.
191            
192        Returns
193        -------
194        psf : 2D-numpy array
195            The array represents estimate of experimental PSF.
196        '''
197        # (0) Prepare variables
198        file_counter = 0
199        R = SDATA.detector.upscale
200        det_size = SDATA.detector.detector_size
201        psf_size = DIFFIMAGES.psfsize
202        # (1) Prepare empty array for PSF calculation
203        psf = np.zeros((det_size*R, det_size*R), dtype=np.float32)
204        if psf_size:
205            psf =  np.zeros((psf_size*R, psf_size*R), dtype=np.float32)
206        else:
207            psf =  np.zeros((det_size*R, det_size*R), dtype=np.float32)
208        # (2) Go through the files and calculate average PSF out of them
209        for index,datafile in df.iterrows():
210            # (a) Read datafile to array and increase file_counter
211            file_to_read = SDATA.data_dir.joinpath(datafile.DatafileName)
212            arr = stemdiff.io.Datafiles.read(SDATA, file_to_read)
213            file_counter += 1
214            # (b) Rescale array
215            arr = stemdiff.io.Arrays.rescale(arr, R, order=3)
216            # (c) Remove edges if psf_size is given
217            if psf_size:
218                xc,yc = stemdiff.io.Arrays.find_center(arr,
219                    csquare = DIFFIMAGES.csquare*R,
220                    cintensity = DIFFIMAGES.cintensity)
221                arr = stemdiff.io.Arrays.remove_edges(arr,
222                    rsize = psf_size*R, xc=round(xc), yc=round(yc))
223                psf += arr
224            else:
225                psf += arr
226        # (3) Calculate final experimental PSF
227        # (divide sum by no_of_summed_files in order to get reasonable values
228        psf = np.round(psf/file_counter).astype(np.uint16)
229        # (4) Return final experimental PSF
230        return(psf)

234    def get_psf(arr, psf_size, circular=False):
235        '''
236        Get PSF of type2 = estimated small central region of datafile.
237
238        Parameters
239        ----------
240        arr : 2D-numpy array
241            The array/datafile, from which the PSF is to be determined.
242            The array is a square with geometrical center = intensity center.
243            See *Technical notes* below for more details and consequences.
244        psf_size : int
245            The size/diameter of PSF function to be determined.
246        circular : bool, optional, default is False
247            If True, apply circular mask - change square PSF to circular PSF.
248            
249        Returns
250        -------
251        psf : 2D-numpy array
252            The array represents estimate of experimental PSF.
253
254        Technical notes
255        --------------
256        In our algorithm, we send a square centered array to this function.
257        This means that intensity center = geometric center of the array.
258        Consequence: we do not have to determine intensity center
259        and waste time - it is enough to calculate geometrical center.
260        '''
261        # Get center
262        # (just geometrical center as explained in the docstring
263        xc = yc = arr.shape[0]//2
264        # Remove edges
265        # (leave only central square with psf_size 
266        psf = stemdiff.io.Arrays.remove_edges(arr, psf_size, xc,yc)
267        # Remove background
268        # (PSF is from the center, it "sits" on some background intensity
269        # !!! Convert PSF from uint to float - uint fails for negative values!
270        psf = psf.astype(np.float32)
271        psf = psf - np.min(psf)
272        # Convert square PSF to circular PSF if requested
273        if circular:
274            # Apply mask
275            psf = psf * circular_mask(psf_size, psf_size)
276            # Remove the remnants of background
277            # (after removing edges, array values may be slightly above zero!
278            # (1) We subtract the 2nd lowest value - the lowest are the zeros
279            the_second_lowest_value = np.min(psf[psf != np.min(psf)])
280            psf = psf - the_second_lowest_value
281        # All negative values shoud go back above zero!
282        # (the negative values may occur due to various recalcs and roundings
283        # (they are very dangerous, leading to unwanted side-effects and errors
284        psf = np.where(psf < 0, 0, psf)
285        # Return final PSF
286        return(psf)

290    def get_psf(arr, psf_size=100, cake=30, subtract=False):
291        '''
292        Get PSF of type3 = individual PSF based on cake-method.
293
294        Parameters
295        ----------
296        arr : 2D-numpy array
297            The array/datafile, from which the PSF is to be determined.
298            The array is a square with geometrical center = intensity center.
299            See *Technical notes* below for more details and consequences.
300        psf_size : int, optional, default
301            The size/diameter of PSF function to be determined.
302        cake : int
303            Size of cake-piece in degrees.
304        subtract : bool, optional, default is False
305            If True, prepare PSF with the same size as scattering pattern.
306            In fact, this function *does not* subtract anything.
307            It can *prepare PSF suitable for subtracting* if required.
308            The trick is that a PSF suitable for subtracting
309            must have the the same size as the original image = arr.
310            
311        Returns
312        -------
313        psf : 2D-numpy array
314            The array represents estimate of experimental PSF.
315
316        Technical notes
317        --------------
318        In our algorithm, we send a square centered array to this function.
319        This means that intensity center = geometric center of the array.
320        Consequence: we do not have to determine intensity center
321        and waste time - it is enough to calculate geometrical center.
322        '''
323           
324        if subtract == True:     
325            # PSF in the same size as image, converted to circular PSF
326            arr  = arr * circular_mask(arr.shape[0], arr.shape[1])     
327        else:   
328            # Picking the PSF from center of image
329            od = int(arr.shape[0]/2)-psf_size//2
330            do = int(arr.shape[0]/2)+psf_size//2
331            arr = arr[od:do,od:do]
332            # Convert to circular PSF
333            arr = arr * circular_mask(psf_size, psf_size)  
334
335        # Initial rotation angle
336        angle = 0
337        psf_multilevel = np.zeros((arr.shape[0], arr.shape[1], cake)) 
338        # Perform roation - PSF smooting
339        for j in range(0,cake):
340            # Storing all rotated diffraction patterns
341            psf_multilevel[:,:,j] = scipy.ndimage.rotate(
342                arr, angle=angle, reshape = False) 
343            angle += 1
344    
345        psf = np.zeros((arr.shape[0],arr.shape[1]))
346        # Pixels selection without peak 
347        for i in range(arr.shape[0]):
348            for ii in range(arr.shape[1]):
349                sig_sorted = np.sort(psf_multilevel[i,ii,:], axis=None)
350                # Prepared for np.mean and interval [0:x]
351                psf[i,ii] = np.median(sig_sorted[:]) 
352        
353        # Important safety insertion
354        # (due to various recalcs and roundings, psf may go below 0
355        # (negative psf values result in many unwanted side effects and errors!
356        psf = np.where(psf < 0, 0, psf)
357        
358        return(psf)

237        Get PSF of type2 = estimated small central region of datafile.
238
239        Parameters
240        ----------
241        arr : 2D-numpy array
242            The array/datafile, from which the PSF is to be determined.
243            The array is a square with geometrical center = intensity center.
244            See *Technical notes* below for more details and consequences.
245        psf_size : int
246            The size/diameter of PSF function to be determined.
247        circular : bool, optional, default is False
248            If True, apply circular mask - change square PSF to circular PSF.
249            
250        Returns
251        -------
252        psf : 2D-numpy array
253            The array represents estimate of experimental PSF.
254
255        Technical notes
256        --------------
257        In our algorithm, we send a square centered array to this function.
258        This means that intensity center = geometric center of the array.
259        Consequence: we do not have to determine intensity center
260        and waste time - it is enough to calculate geometrical center.
261        '''
262        # Get center
263        # (just geometrical center as explained in the docstring
264        xc = yc = arr.shape[0]//2
265        # Remove edges
266        # (leave only central square with psf_size 
267        psf = stemdiff.io.Arrays.remove_edges(arr, psf_size, xc,yc)
268        # Remove background
269        # (PSF is from the center, it "sits" on some background intensity
270        # !!! Convert PSF from uint to float - uint fails for negative values!
271        psf = psf.astype(np.float32)
272        psf = psf - np.min(psf)
273        # Convert square PSF to circular PSF if requested
274        if circular:
275            # Apply mask
276            psf = psf * circular_mask(psf_size, psf_size)
277            # Remove the remnants of background
278            # (after removing edges, array values may be slightly above zero!
279            # (1) We subtract the 2nd lowest value - the lowest are the zeros
280            the_second_lowest_value = np.min(psf[psf != np.min(psf)])
281            psf = psf - the_second_lowest_value
282        # All negative values shoud go back above zero!
283        # (the negative values may occur due to various recalcs and roundings
284        # (they are very dangerous, leading to unwanted side-effects and errors
285        psf = np.where(psf < 0, 0, psf)
286        # Return final PSF
287        return(psf)

295        Parameters
296        ----------

298            The array/datafile, from which the PSF is to be determined.
299            The array is a square with geometrical center = intensity center.
300            See *Technical notes* below for more details and consequences.
301        psf_size : int, optional, default
302            The size/diameter of PSF function to be determined.
303        cake : int
304            Size of cake-piece in degrees.
305        subtract : bool, optional, default is False
306            If True, prepare PSF with the same size as scattering pattern.
307            In fact, this function *does not* subtract anything.
308            It can *prepare PSF suitable for subtracting* if required.
309            The trick is that a PSF suitable for subtracting
310            must have the the same size as the original image = arr.
311            
312        Returns
313        -------
314        psf : 2D-numpy array
315            The array represents estimate of experimental PSF.
316
317        Technical notes
318        --------------
319        In our algorithm, we send a square centered array to this function.
320        This means that intensity center = geometric center of the array.
321        Consequence: we do not have to determine intensity center
322        and waste time - it is enough to calculate geometrical center.
323        '''
324           
325        if subtract == True:     
326            # PSF in the same size as image, converted to circular PSF
327            arr  = arr * circular_mask(arr.shape[0], arr.shape[1])     
328        else:   
329            # Picking the PSF from center of image
330            od = int(arr.shape[0]/2)-psf_size//2
331            do = int(arr.shape[0]/2)+psf_size//2
332            arr = arr[od:do,od:do]
333            # Convert to circular PSF
334            arr = arr * circular_mask(psf_size, psf_size)  
335
336        # Initial rotation angle
337        angle = 0
338        psf_multilevel = np.zeros((arr.shape[0], arr.shape[1], cake)) 
339        # Perform roation - PSF smooting
340        for j in range(0,cake):
341            # Storing all rotated diffraction patterns
342            psf_multilevel[:,:,j] = scipy.ndimage.rotate(
343                arr, angle=angle, reshape = False) 
344            angle += 1
345    
346        psf = np.zeros((arr.shape[0],arr.shape[1]))
347        # Pixels selection without peak 
348        for i in range(arr.shape[0]):
349            for ii in range(arr.shape[1]):
350                sig_sorted = np.sort(psf_multilevel[i,ii,:], axis=None)
351                # Prepared for np.mean and interval [0:x]
352                psf[i,ii] = np.median(sig_sorted[:])

356        # (negative psf values result in many unwanted side effects and errors!
357        psf = np.where(psf < 0, 0, psf)
